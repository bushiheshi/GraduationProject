    elif choice == '3':
        num1 = float(input("请输入第一个数字："))
        num2 = float(input("请输入第二个数字："))
        result = num1 * num2

        num1 = float(input("请输入第一个数字："))
        num2 = float(input("请输入第二个数字："))
