        num1 = float(input("请输入第一个数字："))
        num2 = float(input("请输入第二个数字："))
        result = num1 * num2
        print(f"结果是：{result}")
    elif choice == '4':
        num1 = float(input("请输入第一个数字："))
        num2 = float(input("请输入第二个数字："))

            result = num1 / num2
            print(f"结果是：{result}")
        else:
            print("错误：除数不能为零")
